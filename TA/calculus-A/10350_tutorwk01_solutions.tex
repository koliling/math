\nd {\bf 1.} Let $f(x)=\dfrac{1}{x}$. Assuming that $h\neq 0$, find and simplify \quad$\dfrac{f(x+h)-f(x)}{h}$.

\vskip .2in

\begin{align*}
\dfrac{f(x+h)-f(x)}{h} &= \dfrac{\dfrac{1}{x+h} - \dfrac{1}{x}}{h} \\
&= \dfrac{1}{h} \left( \dfrac{x}{x(x+h)} - \dfrac{x+h}{x(x+h)} \right) \\
&= \dfrac{1}{h} \left( \dfrac{-h}{x^2+xh} \right) \\
&= \dfrac{-1}{x^2+xh} \text{ (accepted solution)}\\
&= \dfrac{-1}{x(x+h)} \text{ (accepted solution)}
\end{align*}

\begin{align*}
 4^{x-2} &= 8 \\
(2^2)^{(x-2)}& = 2^3 \\
 2^{2x-4} &= 2^3 \\
 2x-4 &= 3 \\
2x &= 7 \\
 x&=\dfrac{7}{2}
\end{align*}

\nd {\bf 5.} Find and simplify the expression $\dfrac{g(n+1)}{g(n)}$ if $g(n)=\dfrac{2^{n}x^{2n-1}}{n^3}$. 

\begin{align*}
\dfrac{g(n+1)}{g(n)} &= \dfrac{\dfrac{2^{n+1} x^{2(n+1)-1}}{(n+1)^3}}{\dfrac{2^n x^{2n-1}}{n^3}}\\
&= \left( \dfrac{2^{n+1} x^{2(n+1)-1}}{(n+1)^3} \right) \left( \dfrac{n^3}{2^n x^{2n-1}} \right) \\
&= \dfrac{2^{n+1}x^{2n+2-1}n^3}{(n+1)^3 2^n x^{2n-1}} \\
&= \dfrac{2^{(n+1)-n} x^{2n+1-(2n-1)} n^3}{(n+1)^3} \\
&= \dfrac{2x^2 n^3}{(n+1)^3}
\end{align*}
