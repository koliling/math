\section{CW-Complex}
\begin{mydef}
  A \textit{CW-complex} is a space with a filtration
  \begin{equation*}
    X^{(0)}\subseteq X^{(1)}\subseteq \ldots \subseteq X
  \end{equation*}

    \item $\bigcup\limits_n X^{(n)}=X$
